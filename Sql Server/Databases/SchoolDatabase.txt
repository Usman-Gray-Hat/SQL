create database school
use school

create table tbl_admin
(
ad_id int primary key identity,
ad_name varchar(20) not null,
ad_password varchar(20) not null,
ad_email varchar(40) unique not null
)



create table student
(
std_id int primary key identity,
std_name varchar(20) not null,
std_class varchar(12) not null,
std_gender varchar(10) not null,
std_contact varchar(20) unique not null,
std_email varchar(40) unique not null,
ad_id_fk int foreign key references tbl_admin (ad_id) not null
)



create table fee
(
fee_id int primary key identity,
fee_amount int,
std_id_fk int foreign key references student (std_id) not null,
ad_ad_fk int foreign key references tbl_admin (ad_id) not null
)
